-- exadata_io_drilldown_historic.sql
-- Purpose:
--   Historic Exadata + PGA/TEMP drilldown using AWR / DBA_HIST views.
-- Scope:
--   Best for after-the-fact RCA and older time windows.
-- Usage:
--   @exadata_io_drilldown_historic.sql
-- Inputs:
--   start_time : YYYY-MM-DD HH24:MI:SS
--   end_time   : YYYY-MM-DD HH24:MI:SS
--   top_n      : number of SQLs to drill into

set pagesize 5000
set linesize 300
set trimspool on
set tab off
set verify off
set feedback on
set serveroutput on size unlimited
alter session set nls_date_format='YYYY-MM-DD HH24:MI:SS';

prompt
prompt Usage:
prompt   @exadata_io_drilldown_historic.sql
prompt Then enter:
prompt   Start time = YYYY-MM-DD HH24:MI:SS
prompt   End time   = YYYY-MM-DD HH24:MI:SS
prompt   Top SQL count
prompt
prompt Notes:
prompt   - Uses DBA_HIST_ACTIVE_SESS_HISTORY / DBA_HIST_SQLSTAT / DBA_HIST_SQL_PLAN
prompt   - Best for historic RCA
prompt   - For active incidents, use exadata_io_drilldown_current.sql
prompt

accept p_start char prompt 'Start time (YYYY-MM-DD HH24:MI:SS): '
accept p_end   char prompt 'End time   (YYYY-MM-DD HH24:MI:SS): '
accept p_topn  char default '10' prompt 'Top SQL count to drill into [10]: '

prompt ============================================================================
prompt HISTORIC EXADATA IO + PGA/TEMP DRILLDOWN
prompt START: &p_start
prompt END  : &p_end
prompt TOP N: &p_topn
prompt ============================================================================

column event format a40
column wait_class format a18
column sql_id format a13

prompt
prompt [1] WINDOW-LEVEL TOP SQL BY HISTORIC ASH SAMPLES
WITH top_sql AS (
  SELECT ash.sql_id,
         COUNT(*) AS samples,
         ROUND(COUNT(*) * 100 / SUM(COUNT(*)) OVER (), 2) AS pct
  FROM   dba_hist_active_sess_history ash
  WHERE  ash.dbid = (SELECT dbid FROM v$database)
    AND  ash.sample_time BETWEEN TO_DATE('&p_start','YYYY-MM-DD HH24:MI:SS')
                             AND TO_DATE('&p_end','YYYY-MM-DD HH24:MI:SS')
    AND  ash.sql_id IS NOT NULL
  GROUP  BY ash.sql_id
)
SELECT *
FROM (
  SELECT sql_id, samples, pct
  FROM   top_sql
  ORDER  BY samples DESC
)
WHERE ROWNUM <= TO_NUMBER('&p_topn');

prompt
prompt [2] WINDOW-LEVEL HISTORIC WAIT PROFILE
SELECT NVL(ash.event,'ON CPU') AS event,
       ash.wait_class,
       COUNT(*) AS samples,
       ROUND(COUNT(*) * 100 / SUM(COUNT(*)) OVER (), 2) AS pct
FROM   dba_hist_active_sess_history ash
WHERE  ash.dbid = (SELECT dbid FROM v$database)
  AND  ash.sample_time BETWEEN TO_DATE('&p_start','YYYY-MM-DD HH24:MI:SS')
                           AND TO_DATE('&p_end','YYYY-MM-DD HH24:MI:SS')
GROUP  BY NVL(ash.event,'ON CPU'), ash.wait_class
ORDER  BY samples DESC;

prompt
prompt [3] HISTORIC TOP SQL BY TEMP / EXADATA / DIRECT PATH PROFILE
WITH base AS (
  SELECT ash.sql_id,
         SUM(CASE WHEN ash.event IN ('direct path read temp','direct path write temp') THEN 1 ELSE 0 END) AS temp_samples,
         SUM(CASE WHEN ash.event LIKE 'cell%' THEN 1 ELSE 0 END) AS cell_samples,
         SUM(CASE WHEN ash.event LIKE 'direct path%' THEN 1 ELSE 0 END) AS dp_samples,
         COUNT(*) AS total_samples
  FROM   dba_hist_active_sess_history ash
  WHERE  ash.dbid = (SELECT dbid FROM v$database)
    AND  ash.sample_time BETWEEN TO_DATE('&p_start','YYYY-MM-DD HH24:MI:SS')
                             AND TO_DATE('&p_end','YYYY-MM-DD HH24:MI:SS')
    AND  ash.sql_id IS NOT NULL
  GROUP  BY ash.sql_id
)
SELECT *
FROM (
  SELECT sql_id,
         total_samples,
         temp_samples,
         cell_samples,
         dp_samples,
         ROUND(temp_samples * 100 / NULLIF(total_samples,0), 2) AS temp_pct,
         ROUND(cell_samples * 100 / NULLIF(total_samples,0), 2) AS cell_pct,
         ROUND(dp_samples   * 100 / NULLIF(total_samples,0), 2) AS dp_pct
  FROM   base
  ORDER  BY total_samples DESC
)
WHERE ROWNUM <= TO_NUMBER('&p_topn');

prompt
prompt [4] PER-SQL DETAIL WRAPPER LINES
declare
  cursor c_sql is
    SELECT sql_id
    FROM (
      SELECT ash.sql_id, COUNT(*) AS samples
      FROM   dba_hist_active_sess_history ash
      WHERE  ash.dbid = (SELECT dbid FROM v$database)
        AND  ash.sample_time BETWEEN TO_DATE('&p_start','YYYY-MM-DD HH24:MI:SS')
                                 AND TO_DATE('&p_end','YYYY-MM-DD HH24:MI:SS')
        AND  ash.sql_id IS NOT NULL
      GROUP  BY ash.sql_id
      ORDER  BY samples DESC
    )
    WHERE ROWNUM <= TO_NUMBER('&p_topn');
begin
  for r in c_sql loop
    dbms_output.put_line('@exadata_io_sql_detail_historic.sql '||r.sql_id||' "'||'&p_start'||'" "'||'&p_end'||'"');
  end loop;
end;
